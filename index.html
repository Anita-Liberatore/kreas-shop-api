Integration

Fetch the data

Drop this into your Vue.js component and start building. Works from any HTTP client — no CORS issues. 

const API = 'https://raw.githubusercontent.com/
  Anita-Liberatore/kreas-shop-api/master/products.json'

const getProducts = async () => {
  const res = await fetch(API)
  return res.json() // Array of 24 products
}
// Vue 3 — Composition API
import { ref, onMounted } from 'vue'

const products = ref([])

onMounted(async () => {
  const res = await fetch(
    'https://raw.githubusercontent.com/
    Anita-Liberatore/kreas-shop-api/master/products.json'
  )
  products.value = await res.json()
})
// Axios
import axios from 'axios'

const { data: products } = await axios.get(
  'https://raw.githubusercontent.com/
  Anita-Liberatore/kreas-shop-api/master/products.json'
)
// products → Array of 24 items
